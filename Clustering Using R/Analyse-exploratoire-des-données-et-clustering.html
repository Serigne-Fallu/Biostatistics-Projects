h2 <- cutree(h, k = 2)
h2cols <- c("orangered","blue")[h2]

hclust_fun <- function(x){
    f <- hclust(x, method = "ward.D2");
    return(f)
}

heatmap(
    as.matrix(mouse_exp),
    col = brewer.pal("Blues",n=8),
    hclustfun = hclust_fun,
    RowSideColors = h2cols, 
    ColSideColors = c(
        rep("darkgreen",3),
        rep("deeppink2",3)
    )
)
